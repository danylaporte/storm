15/// A trait that wrap the ProviderFactory to be able to use it in a Box<Any> trait object context.
16trait AnyFactory: Send + Sync + 'static {
17    fn create(&self) -> BoxFuture<'_, Result<CastProvider>>;
18}

20/// Wrap a ProviderFactory trait to be able to use it in a Box<Any> trait object context.
21/// This struct implement AnyFactory which can be use in a trait object.
22struct Factory<F, P> {
23    _provider: PhantomData<P>,
24    factory: F,
25}
26
27impl<F, P> Factory<F, P> {
28    fn new(factory: F) -> Self {
29        Self {
30            _provider: PhantomData,
31            factory,
32        }
33    }
34}
35
36impl<F, P> AnyFactory for Factory<F, P>
37where
38    F: ProviderFactory<Provider = P>,
39    P: Provider,
40{
41    fn create(&self) -> BoxFuture<'_, Result<CastProvider>> {
42        Box::pin(async move { Ok(CastProvider::new(self.factory.create_provider().await?)) })
43    }
44}

46/// A dependency container to be able to instantiate and provide connection to databases.
47///
48/// A database provider can be named and have a type.
49pub struct ProviderContainer {
50    last_gc: u64,
51    lock: Mutex<()>,
52    lru: Lru,
53    records: Vec<Rec>,
54}
55
56impl ProviderContainer {
57    pub fn new() -> Self {
58        Self::default()
59    }
60
61    fn find_index(&self, type_id: TypeId, name: &str) -> std::result::Result<usize, usize> {
62        self.records.binary_search_by_key(&(type_id, name), rec_key)
63    }
64
65    fn find_record<'a>(&'a self, type_id: TypeId, name: &str) -> Result<&'a Rec> {
66        match self.find_index(type_id, name) {
67            Ok(index) => self.records.get(index).ok_or(Error::ProviderNotFound),
68            Err(_) => Err(Error::ProviderNotFound),
69        }
70    }
71
72    pub(crate) async fn gate(&self) -> MutexGuard<'_, ()> {
73        self.lock.lock().await
74    }
75
76    /// A method to garbage collect all unused provider. This is intended to close database
77    /// connections and release resources.
78    pub fn gc(&mut self) {
79        let last_gc = self.last_gc;
80        let new_gc = *self.lru.get_mut();
81
82        if last_gc != new_gc {
83            for r in &mut self.records {
84                if r.provider
85                    .get_mut()
86                    .is_some_and(|r| *r.lru.get_mut() <= last_gc)
87                {
88                    r.provider.take();
89                }
90            }
91
92            self.last_gc = new_gc;
93        }
94    }
95
96    /// Gets or creates a database provider that have been previously registered with
97    /// the `register` method.
98    pub fn provide<'a, P: Provider>(&'a self, name: &'a str) -> BoxFuture<'a, Result<&'a P>> {
99        Box::pin(async move {
100            let type_id = TypeId::of::<P>();
101            let rec = self.find_record(type_id, name)?;
102            let castable = rec.get_or_init(&self.lru).await?;
103
104            castable.downcast().ok_or_else(|| {
105                error!("invalid cast for provider {name}");
106                Error::Internal
107            })
108        })
109    }
110
111    pub(super) fn providers(&self) -> impl Iterator<Item = &'_ dyn Provider> {
112        self.records
113            .iter()
114            .filter_map(|r| Some(r.get()?.provider()))
115    }
116
117    /// Register a provider factory that creates provider on demand. A provider can be named.
118    pub fn register<F: ProviderFactory>(&mut self, name: impl Into<Box<str>>, factory: F) {
119        let rec = Rec {
120            factory: Box::new(Factory::new(factory)),
121            name: name.into(),
122            provider: AsyncOnceCell::new(),
123            type_id: TypeId::of::<F::Provider>(),
124        };
125
126        #[allow(clippy::indexing_slicing)]
127        match self.find_index(rec.type_id, &rec.name) {
128            Ok(index) => self.records[index] = rec,
129            Err(index) => self.records.insert(index, rec),
130        }
131    }
132
133    pub fn transaction(&self) -> TransactionProvider<'_> {
134        TransactionProvider(self)
135    }
136}
137
138impl Default for ProviderContainer {
139    fn default() -> Self {
140        Self {
141            last_gc: 0,
142            lock: Mutex::new(()),
143            lru: AtomicU64::new(1), // starting at 1 because the garbage collector start at 0.
144            records: Vec::new(),
145        }
146    }
147}

163struct Rec {
164    factory: Box<dyn AnyFactory>,
165    name: Box<str>,
166    provider: AsyncOnceCell<ProviderRec>,
167    type_id: TypeId,
168}
169
170impl Rec {
171    fn get(&self) -> Option<&CastProvider> {
172        Some(&self.provider.get()?.cast_provider)
173    }
174
175    async fn get_or_init<'a>(&'a self, lru: &'a Lru) -> Result<&'a CastProvider> {
176        let provider_rec = self
177            .provider
178            .get_or_try_init::<_, Error>(async {
179                Ok(ProviderRec::new(self.factory.create().await?))
180            })
181            .await?;
182
183        provider_rec.lru.store(lru.fetch_add(1, Relaxed), Relaxed);
184
185        Ok(&provider_rec.cast_provider)
186    }
187}

189fn rec_key(rec: &Rec) -> (TypeId, &str) {
190    (rec.type_id, &rec.name)
