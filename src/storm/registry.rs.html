1use chrono::NaiveDateTime;
2use std::cell::{Cell, UnsafeCell};
3
4pub(crate) struct InitCell<T>(UnsafeCell<T>);
5
6impl<T> InitCell<T> {
7    #[inline]
8    pub(crate) const fn new(val: T) -> Self {
9        Self(UnsafeCell::new(val))
10    }
11
12    #[inline]
13    pub fn get(&self) -> &T {
14        unsafe { &*self.0.get() }
15    }
16
17    #[allow(clippy::mut_from_ref)]
18    #[inline]
19    pub fn get_mut(&self) -> &mut T {
20        check_can_register();
21        unsafe { &mut *self.0.get() }
22    }
23}
24
25unsafe impl<T: Sync> Send for InitCell<T> {}
26unsafe impl<T: Sync> Sync for InitCell<T> {}

28/// Private : For macro only.
29#[doc(hidden)]
30#[linkme::distributed_slice]
31pub static __REGISTRATION: [fn()];
32
33pub(crate) fn check_can_register() {
34    assert!(
35        IN_REGISTRATION.get(),
36        "must be call inside the registration phase."
37    );
38}
39
40pub(crate) fn perform_registration() {
41    static O: parking_lot::Once = parking_lot::Once::new();
42
43    O.call_once(|| {
44        IN_REGISTRATION.set(true);
45
46        for f in __REGISTRATION {
47            f();
48        }
49
50        IN_REGISTRATION.set(false);
51    });
52}
53
54thread_local! {
55    static IN_REGISTRATION: Cell<bool> = const { Cell::new(false) };
56}
