7pub(crate) fn indexing(item: Item) -> TokenStream {
8    match &item {
9        Item::Fn(f) => indexing_fn(f),
10        _ => Error::new(item.span(), "Only function is supported.").to_compile_error(),
11    }
12}
13
14fn gc(index_name: &Ident, index_ty: &Type) -> (TokenStream, TokenStream) {
15    (
16        quote! {
17            impl storm::Gc for #index_name {
18                const SUPPORT_GC: bool = <#index_ty as storm::Gc>::SUPPORT_GC;
19
20                fn gc(&mut self, ctx: &storm::GcCtx) {
21                    self.0.gc(ctx);
22                }
23            }
24        },
25        quote! {
26            if <#index_ty as storm::Gc>::SUPPORT_GC {
27                storm::gc::collectables::register(|ctx| ctx.index_gc::<#index_name>());
28            }
29        },
30    )
31}

33fn indexing_fn(f: &ItemFn) -> TokenStream {
34    let vis = &f.vis;
35    let name = &f.sig.ident;
36    let name_str = &name.to_string().to_pascal_case();
37    let index_name = Ident::new(name_str, name.span());
38    let index_name_lit = LitStr::new(name_str, name.span());
39
40    let ty = match &f.sig.output {
41        ReturnType::Type(_, t) => t,
42        ReturnType::Default => {
43            return Error::new(f.sig.output.span(), "Index must have a return value.")
44                .to_compile_error()
45        }
46    };
47
48    let mut args = Vec::new();
49
50    for arg in &f.sig.inputs {
51        match arg {
52            FnArg::Typed(t) => args.push(t),
53            FnArg::Receiver(r) => {
54                return Error::new(r.span(), "Self is not expected here.").to_compile_error()
55            }
56        };
57    }
58
59    let mut as_ref_decl = Vec::new();
60    let mut as_ref_decl_async = Vec::new();
61    let mut as_ref_args = Vec::new();
62    let mut as_ref_tag = Vec::new();
63    let mut as_ref_async_wheres = quote!();
64    let mut as_ref_wheres = quote!();
65    let mut deps = Vec::new();
66
67    for (index, arg) in args.iter().enumerate() {
68        let ty = unref(&arg.ty);
69        let ident = Ident::new(&format!("var{index}"), arg.span());
70
71        if ty.is_storm_ctx() {
72            as_ref_decl.push(quote!(let #ident = self.ctx;));
73            as_ref_decl_async.push(quote!(let #ident = self;));
74            as_ref_args.push(ident);
75        } else {
76            as_ref_decl.push(quote!(let #ident = self.as_ref();));
77            as_ref_decl_async.push(
78                quote!(let #ident = storm::tri!(storm::AsRefAsync::as_ref_async(self).await);),
79            );
80            as_ref_tag.push(quote!(storm::Tag::tag(#ident)));
81            as_ref_args.push(ident);
82            deps.push(quote!(<#ty as storm::Accessor>::register_deps(<#index_name as storm::Accessor>::clear);));
83
84            let ref_async = quote!(storm::AsRefAsync<#ty>);
85            let ref_sync = quote!(AsRef<#ty>);
86
87            if as_ref_wheres.is_empty() {
88                as_ref_async_wheres = quote!(where Self: #ref_async);
89                as_ref_wheres = quote!(where Self: #ref_sync);
90            } else {
91                as_ref_async_wheres = quote!(#as_ref_async_wheres + #ref_async);
92                as_ref_wheres = quote!(#as_ref_wheres + #ref_sync);
93            };
94        }
95    }
96
97    let get_or_init = quote!({
98        let _ = tracing::span!(tracing::Level::DEBUG, <#index_name as storm::CtxTypeInfo>::NAME, obj = storm::OBJ_INDEX, ev = storm::EV_CREATED).entered();
99        #index_name(#name(#(#as_ref_args,)*), storm::version_tag::combine(&[#(#as_ref_tag,)*]))
100    });
101
102    let (gc, gc_collect) = gc(&index_name, ty);
103
104    quote! {
105        #vis struct #index_name(#ty, storm::VersionTag);
106
107        impl storm::Accessor for #index_name {
108            #[allow(non_camel_case_types)]
109            #[inline]
110            fn var() -> storm::TblVar<Self> {
111                storm::attached::var!(T: #index_name, storm::vars::Tbl);
112
113                #[static_init::dynamic]
114                static R: () = {
115                    #(#deps)*
116                    #gc_collect
117                };
118
119                *T
120            }
121
122            #[inline]
123            fn deps() -> &'static storm::Deps {
124                static DEPS: storm::Deps = storm::parking_lot::RwLock::new(Vec::new());
125                &DEPS
126            }
127        }
128
129        impl std::ops::Deref for #index_name {
130            type Target = #ty;
131
132            #[inline]
133            fn deref(&self) -> &Self::Target {
134                &self.0
135            }
136        }
137
138        impl<'a, L> AsRef<#index_name> for storm::CtxLocks<'a, L> #as_ref_wheres {
139            fn as_ref(&self) -> &#index_name {
140                let var = <#index_name as storm::Accessor>::var();
141                let ctx = self.ctx.vars();
142
143                if let Some(v) = ctx.get(var) {
144                    return v;
145                }
146
147                #(#as_ref_decl)*
148
149                if let Some(v) = ctx.get(var) {
150                    return v;
151                }
152
153                let instant = std::time::Instant::now();
154                let r = ctx.get_or_init(var, move || #get_or_init);
155                storm::debug_index_get_or_init_elapsed(instant, #index_name_lit);
156                r
157            }
158        }
159
160        impl storm::AsRefAsync<#index_name> for storm::Ctx #as_ref_async_wheres {
161            fn as_ref_async(&self) -> storm::BoxFuture<'_, storm::Result<&'_ #index_name>> {
162                let var = <#index_name as storm::Accessor>::var();
163
164                Box::pin(async move {
165                    let ctx = self.vars();
166
167                    if let Some(v) = ctx.get(var) {
168                        return Ok(v);
169                    }
170
171                    #(#as_ref_decl_async)*
172
173                    if let Some(v) = ctx.get(var) {
174                        return Ok(v);
175                    }
176
177                    let _gate = self.provider().gate(#index_name_lit).await;
178                    Ok(ctx.get_or_init(var, || #get_or_init))
179                })
180            }
181        }
182
183        impl storm::Tag for #index_name {
184            fn tag(&self) -> storm::VersionTag {
185                self.1
186            }
187        }
188
189        impl storm::CtxTypeInfo for #index_name {
190            const NAME: &'static str = #index_name_lit;
191        }
192
193        #gc
194
195        #f
196    }
197}

199fn unref(t: &Type) -> &Type {
200    match t {
201        Type::Reference(r) => unref(&r.elem),
202        _ => t,
203    }
