16fn implement(input: &DeriveInput) -> Result<TokenStream, TokenStream> {
17    let vis = &input.vis;
18    let entity = &input.ident;
19    let args = TypeArgs::from_derive_input(input).map_err(|e| e.write_errors())?;
20
21    let entity_name = entity.to_string();
22    let table_name = entity_name.to_plural();
23    let table_name_lit = LitStr::new(&entity_name, entity.span());
24    let table_alias = Ident::new(&table_name, entity.span());
25
26    let init_tbl_fn = Ident::new(
27        &format!("__ctx_init_{}", entity_name.to_snake_case()),
28        entity.span(),
29    );
30
31    let coll_ty = args.collection.ty(entity);
32    let (gc, gc_collect) = gc(input)?;
33
34    Ok(quote! {
35        #vis type #table_alias = #coll_ty;
36
37        impl storm::EntityAccessor for #entity {
38            type Tbl = #table_alias;
39
40            #[inline]
41            fn applied() -> &'static storm::AppliedEvent<Self> {
42                static E: storm::AppliedEvent<#entity> = storm::AppliedEvent::new();
43                &E
44            }
45
46            #[inline]
47            fn cleared() -> &'static storm::ClearEvent {
48                static E: storm::ClearEvent = storm::ClearEvent::new();
49                &E
50            }
51
52            #[inline]
53            fn removed() -> &'static storm::RemovedEvent<Self> {
54                static E: storm::RemovedEvent<#entity> = storm::RemovedEvent::new();
55                &E
56            }
57
58            #[inline]
59            fn removing() -> &'static storm::RemovingEvent<Self> {
60                static E: storm::RemovingEvent<#entity> = storm::RemovingEvent::new();
61                &E
62            }
63
64            #[inline]
65            fn tbl_var() -> storm::CtxVar<Self::Tbl> {
66                storm::extobj::extobj!(
67                    impl storm::CtxExt {
68                        V: std::sync::OnceLock<#table_alias>,
69                    },
70                    crate_path = storm::extobj
71                );
72
73                *V
74            }
75
76            #[inline]
77            fn touched() -> &'static storm::TouchedEvent {
78                static E: storm::TouchedEvent = storm::TouchedEvent::new();
79                &E
80            }
81
82            #[inline]
83            fn upserted() -> &'static storm::UpsertedEvent<Self> {
84                static E: storm::UpsertedEvent<#entity> = storm::UpsertedEvent::new();
85                &E
86            }
87
88            #[inline]
89            fn upserting() -> &'static storm::UpsertingEvent<Self> {
90                static E: storm::UpsertingEvent<#entity> = storm::UpsertingEvent::new();
91                &E
92            }
93        }
94
95        impl storm::CtxTypeInfo for #entity {
96            const NAME: &'static str = #table_name_lit;
97        }
98
99        #[storm::register]
100        fn #init_tbl_fn() {
101            storm::__register_apply(#table_alias::__apply_log, storm::ApplyOrder::Table);
102            #gc_collect
103        }
104
105        #gc
106    })
107}

109fn gc(input: &DeriveInput) -> Result<(TokenStream, TokenStream), TokenStream> {
110    let fields = input.fields()?;
111    let ident = &input.ident;
112    let types = fields.iter().map(|f| &f.ty);
113    let fields = fields.iter().map(|f| &f.ident);
114
115    Ok((
116        quote! {
117            impl storm::Gc for #ident {
118                const SUPPORT_GC: bool = #(<#types as storm::Gc>::SUPPORT_GC ||)* false;
119
120                fn gc(&mut self) {
121                    #(storm::Gc::gc(&mut self.#fields);)*
122                }
123            }
124        },
125        quote! {
126            if <#ident as storm::Gc>::SUPPORT_GC {
127                storm::Ctx::on_gc_collect(<#ident as storm::EntityAccessor>::tbl_gc);
128            }
129        },
130    ))
131}

133#[derive(Clone, Copy, Debug, Default, Eq, FromMeta, PartialEq)]
134enum Collection {
135    HashTable,
136
137    #[default]

141impl Collection {
142    fn ty(&self, entity: &Ident) -> TokenStream {
143        match self {
144            Self::HashTable => quote!(storm::HashTable<#entity>),
145            Self::VecTable => quote!(storm::VecTable<#entity>),
146        }
147    }
148}
