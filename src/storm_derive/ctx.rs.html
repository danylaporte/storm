16fn implement(input: &DeriveInput) -> Result<TokenStream, TokenStream> {
17    let vis = &input.vis;
18    let entity = &input.ident;
19    let args = TypeArgs::from_derive_input(input).map_err(|e| e.write_errors())?;
20
21    let entity_name = entity.to_string();
22    let table_name = entity_name.to_plural();
23    let table_name_lit = LitStr::new(&entity_name, entity.span());
24    let table_alias = Ident::new(&table_name, entity.span());
25
26    let coll_ty = args.collection.ty(entity);
27    let (gc, gc_collect) = gc(input)?;
28
29    Ok(quote! {
30        #vis type #table_alias = #coll_ty;
31
32        impl storm::EntityAccessor for #entity {
33            type Tbl = #table_alias;
34
35            #[allow(non_camel_case_types)]
36            #[inline]
37            fn entity_var() -> storm::TblVar<Self::Tbl> {
38                storm::attached::var!(T: #table_alias, storm::vars::Tbl);
39
40                // Garbage collection static registering
41                #[static_init::dynamic]
42                static G: () = {
43                    #gc_collect
44                };
45
46                *T
47            }
48
49            #[inline]
50            fn entity_deps() -> &'static storm::Deps {
51                static DEPS: storm::Deps = storm::parking_lot::RwLock::new(Vec::new());
52                &DEPS
53            }
54
55            #[inline]
56            fn on_change() -> &'static storm::OnChange<Self> {
57                #[static_init::dynamic]
58                static E: storm::OnChange<#entity> = Default::default();
59                &E
60            }
61
62            #[inline]
63            fn on_changed() -> &'static storm::OnChanged<Self> {
64                #[static_init::dynamic]
65                static E: storm::OnChanged<#entity> = Default::default();
66                &E
67            }
68
69            #[inline]
70            fn on_remove() -> &'static storm::OnRemove<Self> {
71                #[static_init::dynamic]
72                static E: storm::OnRemove<#entity> = Default::default();
73                &E
74            }
75        }
76
77        impl storm::LogAccessor for #entity {
78            #[inline]
79            fn log_var() -> storm::LogVar<storm::Log<Self>> {
80                storm::attached::var!(L: storm::Log<#entity>, storm::vars::Log);
81
82                #[static_init::dynamic]
83                static R: () = storm::register_apply_log::<#entity>();
84
85                *L
86            }
87        }
88
89        impl storm::CtxTypeInfo for #entity {
90            const NAME: &'static str = #table_name_lit;
91        }
92
93        #gc
94    })
95}

97fn gc(input: &DeriveInput) -> Result<(TokenStream, TokenStream), TokenStream> {
98    let fields = input.fields()?;
99    let ident = &input.ident;
100    let types = fields.iter().map(|f| &f.ty);
101    let fields = fields.iter().map(|f| &f.ident);
102
103    Ok((
104        quote! {
105            impl storm::Gc for #ident {
106                const SUPPORT_GC: bool = #(<#types as storm::Gc>::SUPPORT_GC ||)* false;
107
108                #[allow(unused_variables)]
109                fn gc(&mut self, ctx: &storm::GcCtx) {
110                    #(storm::Gc::gc(&mut self.#fields, ctx);)*
111                }
112            }
113        },
114        quote! {
115            if <#ident as storm::Gc>::SUPPORT_GC {
116                storm::gc::collectables::register(|ctx| ctx.tbl_gc::<#ident>());
117            }
118        },
119    ))
120}

122#[derive(Clone, Copy, Debug, Eq, FromMeta, PartialEq)]
123enum Collection {
124    HashTable,
125    VecTable,
126}
127
128impl Collection {
129    fn ty(&self, entity: &Ident) -> TokenStream {
130        match self {
131            Self::HashTable => quote!(storm::HashTable<#entity>),
132            Self::VecTable => quote!(storm::VecTable<#entity>),
133        }
134    }
135}
136
137impl Default for Collection {
138    fn default() -> Self {
139        Self::VecTable
140    }
141}
