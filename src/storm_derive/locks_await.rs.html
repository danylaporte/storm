7pub(crate) fn locks_await(input: &DeriveInput) -> TokenStream {
8    let type_ident = &input.ident;
9    let mut init_fields = Vec::new();
10    let mut as_refs = Vec::new();
11    let mut tags = Vec::new();
12
13    for field in input.fields().expect("fields") {
14        let f_ident = &field.ident;
15        let ty = unref(&field.ty);
16
17        init_fields
18            .push(quote!(#f_ident: storm::tri!(storm::AsRefAsync::as_ref_async(ctx).await),));
19
20        as_refs.push(quote! {
21            impl<'a> AsRef<#ty> for #type_ident<'a> {
22                #[inline]
23                fn as_ref(&self) -> &#ty {
24                    &self.#f_ident
25                }
26            }
27        });
28
29        tags.push(quote!(storm::Tag::tag(&self.#f_ident)));
30    }
31
32    let as_refs = quote!(#(#as_refs)*);
33    let init_fields = quote!(#(#init_fields)*);
34
35    quote! {
36        impl<'a> storm::AsyncTryFrom<'a, &'a storm::Ctx> for #type_ident<'a> {
37            fn async_try_from(ctx: &'a storm::Ctx) -> storm::BoxFuture<'a, storm::Result<#type_ident<'a>>> {
38                let instant = std::time::Instant::now();
39
40                Box::pin(async move {
41                    let r = #type_ident {
42                        #init_fields
43                    };
44
45                    storm::debug_locks_await_elapsed(instant);
46                    Ok(r)
47                })
48            }
49        }
50
51        impl<'a> #type_ident<'a> {
52            pub async fn from_ctx(ctx: &'a storm::Ctx) -> storm::Result<storm::CtxLocks<'a, #type_ident<'a>>> {
53                Ok(storm::CtxLocks {
54                    locks: storm::tri!(storm::AsyncTryFrom::async_try_from(ctx).await),
55                    ctx,
56                })
57            }
58        }
59
60        impl<'a> storm::Tag for #type_ident<'a> {
61            fn tag(&self) -> storm::VersionTag {
62                let mut out = storm::VersionTag::zero();
63                #(out = std::cmp::max(out, #tags);)*
64                out
65            }
66        }
67
68        #as_refs
69    }
70}
