10/// creates a select sql query for each field and keys and generate the
11/// loading code for translated fields.
12pub(super) struct LoadTranslated<'a> {
13    attrs: &'a TypeAttrs,
14    entity: &'a Ident,
15    select: SelectBuilder,
16    fields: Vec<TokenStream>,
17}
18
19impl<'a> LoadTranslated<'a> {
20    pub fn new(entity: &'a Ident, attrs: &'a TypeAttrs) -> Self {
21        Self {
22            attrs,
23            entity,
24            fields: Default::default(),
25            select: SelectBuilder::with_alias("a"),
26        }
27    }
28
29    pub fn add_field(&mut self, field: &Field, column: &str) {
30        let ident = &field.ident;
31        let index = self.select.add_field(column);
32        let read = read_row(index);
33
34        self.fields.push(quote! {
35            let v: Option<&str> = #read;
36            if let Some(v) = v {
37                val.#ident.set(culture, v);
38            }
39        });
40    }
41
42    pub fn to_where_clause(&self) -> TokenStream {
43        let entity = &self.entity;
44
45        match self.fields.is_empty() {
46            false => quote!(+ storm::GetMut<#entity>),
47            true => quote!(),
48        }
49    }
50}
51
52impl ToTokens for LoadTranslated<'_> {
53    fn to_tokens(&self, tokens: &mut TokenStream) {
54        let mut errors = Vec::new();
55
56        if self.select.is_empty() {
57            check_empty(&self.attrs.translate_table, &mut errors);
58            check_empty(&self.attrs.translate_keys, &mut errors);
59        } else {
60            check_required(&self.attrs.translate_table, &mut errors);
61
62            let mut select = self.select.clone();
63            let mut joins = JoinBuilder::default();
64            let mut conds = joins.inner_join(&self.attrs.table, Some("t"));
65
66            let keys = add_keys(self.attrs, &mut conds, &mut select, &mut errors);
67            let culture = read_row(select.add_field("Culture"));
68            let sql = select.to_sql_lit(&self.attrs.translate_table, &self.attrs.where_clause);
69
70            let joins = format!("{{}} {} WHERE {{}}", joins.to_sql());
71            let joins = LitStr::new(&joins, Span::call_site());
72            let entity = self.entity;
73            let fields = &self.fields;
74            let fields = quote!(#(#fields)*);
75
76            tokens.append_all(quote! {
77                const TRANSLATED_SQL: &str = #sql;
78
79                let translated_sql = match sql.is_empty() {
80                    false => format!(#joins, TRANSLATED_SQL, sql),
81                    true => TRANSLATED_SQL.to_string(),
82                };
83
84                let _: storm::provider::LoadDoNothing = storm::tri!(storm_mssql::QueryRows::query_rows(provider, translated_sql, &*params, |row| {
85                    let key: <#entity as storm::Entity>::Key = #keys;
86                    let culture = #culture;
87
88                    if let Some(val) = map.get_mut(&key) {
89                        #fields
90                    }
91
92                    Ok(())
93                }, /*use_transaction*/ true).await);
94            });
95        }
96
97        tokens.append_all(quote!(#(#errors)*));
98    }
99}

101fn add_key(
102    translate_key: &str,
103    key: &str,
104    joins: &mut JoinConditions,
105    select: &mut SelectBuilder,
106    key_ts: &mut Vec<TokenStream>,
107) {
108    let column_index = select.add_field(translate_key);
109
110    key_ts.push(read_row(column_index));
111    joins.add((Some("a"), translate_key), (Some("t"), key));
112}
113
114fn add_keys(
115    type_attrs: &TypeAttrs,
116    joins: &mut JoinConditions,
117    select: &mut SelectBuilder,
118    errors: &mut Vec<TokenStream>,
119) -> TokenStream {
120    let translate_keys = type_attrs.translate_keys(errors);
121    let keys = type_attrs.keys_internal();
122    let mut ts = Vec::new();
123
124    for (translate_key, key) in translate_keys.iter().zip(&keys) {
125        add_key(translate_key, key, joins, select, &mut ts);
126    }
127
128    if keys.len() == 1 {
129        quote!(#(#ts)*)
130    } else {
131        quote!((#(#ts,)*))
132    }
133}
