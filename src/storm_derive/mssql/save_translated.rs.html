16impl<'a> SaveTranslated<'a> {

25    pub fn add_field(&mut self, field: &Field, column: &str) {
26        let ident = &field.ident;
27        let param_index = self.params.add_ts(quote!(&v.#ident.get(culture) as _));
28
29        self.upsert.add_field(column, &param_index.to_string());
30    }

33impl ToTokens for SaveTranslated<'_> {
34    fn to_tokens(&self, tokens: &mut TokenStream) {
35        let mut errors = Vec::new();
36
37        if self.params.is_empty() {
38            check_empty(&self.attrs.translate_table, &mut errors);
39            check_empty(&self.attrs.translate_keys, &mut errors);
40        } else {
41            check_required(&self.attrs.translate_table, &mut errors);
42
43            let mut params = self.params.clone();
44            let mut upsert = self.upsert.clone();
45            let keys = self.attrs.translate_keys(&mut errors);
46
47            add_keys(&keys, &mut params, &mut upsert);
48
49            upsert.add_key("culture", &params.add_ts(quote!(&culture as _)).to_string());
50
51            let sql = upsert.to_sql_lit(&self.attrs.translate_table);
52
53            tokens.append_all(quote! {
54                for &culture in Culture::DB_CULTURES.iter() {
55                    storm::tri!(storm_mssql::Execute::execute(provider, #sql, #params).await);
56                }
57            });
58        }
59
60        tokens.append_all(quote!(#(#errors)*));
61    }
62}

64#[cold]
65fn add_key_many(keys: &[&str], params: &mut ParamsBuilder, builder: &mut UpsertBuilder) {
66    for (index, column) in keys.iter().enumerate() {
67        let i = LitInt::new(&index.to_string(), Span::call_site());
68        add_key_single(column, quote!(&k.#i as _), params, builder);
69    }
70}
71
72fn add_key_single(
73    column: &str,
74    ts: TokenStream,
75    params: &mut ParamsBuilder,
76    builder: &mut UpsertBuilder,
77) {
78    let i = params.add_ts(ts);
79    builder.add_key(column, &i.to_string());
80}
81
82fn add_keys(keys: &[&str], params: &mut ParamsBuilder, builder: &mut UpsertBuilder) {
83    match keys {
84        [k] => add_key_single(k, quote!(k as _), params, builder),
85        _ => add_key_many(keys, params, builder),
86    }
