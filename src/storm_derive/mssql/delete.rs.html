25impl<S> ToTokens for Delete<'_, S>
26where
27    S: AttrsSelector,
28{
29    fn to_tokens(&self, tokens: &mut TokenStream) {
30        let table = S::table(self.attrs);
31
32        if table.is_empty() {
33            return;
34        }
35
36        let mut errors = Vec::new();
37        let mut params = ParamsBuilder::default();
38        let mut delete = DeleteBuilder::default();
39
40        let keys = S::keys(self.attrs, &mut errors);
41
42        add_keys(&keys, &mut params, &mut delete);
43
44        let sql = delete.to_sql_lit(table);
45
46        tokens.append_all(quote! {
47            storm::tri!(storm_mssql::Execute::execute(provider, #sql, #params).await);
48            #(#errors)*
49        });
50    }
51}

53#[cold]
54fn add_key_many(keys: &[&str], params: &mut ParamsBuilder, builder: &mut DeleteBuilder) {
55    for (index, column) in keys.iter().enumerate() {
56        let i = LitInt::new(&index.to_string(), Span::call_site());
57        add_key_single(column, quote!(&k.#i as _), params, builder);
58    }
59}
60
61fn add_key_single(
62    column: &str,
63    ts: TokenStream,
64    params: &mut ParamsBuilder,
65    builder: &mut DeleteBuilder,
66) {
67    let i = params.add_ts(ts);
68    builder.add_key(column, &i.to_string());
69}
70
71fn add_keys(keys: &[&str], params: &mut ParamsBuilder, builder: &mut DeleteBuilder) {
72    match keys {
73        [k] => add_key_single(k, quote!(k as _), params, builder),
74        _ => add_key_many(keys, params, builder),
75    }
76}

78pub(super) trait AttrsSelector {
79    fn keys<'a>(attrs: &'a TypeAttrs, errors: &mut Vec<TokenStream>) -> Vec<&'a str>;
80    fn table(attrs: &TypeAttrs) -> &SpannedValue<String>;
81}
82
83macro_rules! selector {
84    ($t:ty, $keys:ident, $table:ident) => {
85        impl AttrsSelector for $t {
86            fn keys<'a>(attrs: &'a TypeAttrs, errors: &mut Vec<TokenStream>) -> Vec<&'a str> {
87                attrs.$keys(errors)
88            }
89
90            fn table(attrs: &TypeAttrs) -> &SpannedValue<String> {
91                &attrs.$table
92            }
93        }
94    };
95}
