5pub fn register(attr: TokenStream, item: TokenStream) -> TokenStream {
6    let crate_path: Path = match syn::parse::<syn::Meta>(attr) {
7        Ok(syn::Meta::NameValue(nv)) if nv.path.is_ident("crate") => match nv.value {
8            syn::Expr::Path(p) => p.path,
9            _ => {
10                return syn::Error::new(nv.value.span(), "expected path")
11                    .to_compile_error()
12                    .into()
13            }
14        },
15        Ok(_) | Err(_) => syn::parse_quote!(storm), // default / no attr
16    };
17
18    let func = parse_macro_input!(item as ItemFn);
19    if !func.sig.inputs.is_empty() {
20        return syn::Error::new(func.sig.inputs.span(), "function must take no arguments")
21            .to_compile_error()
22            .into();
23    }
24
25    let fn_name = &func.sig.ident;
26    let reg_ident = format_ident!("__{}", fn_name.to_string().to_uppercase());
27
28    quote! {
29        #func
30        #[#crate_path::linkme::distributed_slice(#crate_path::registry::__REGISTRATION)]
31        #[linkme(crate = #crate_path::linkme)]
32        static #reg_ident: fn() = #fn_name;
33    }
34    .into()
35}
