1use crate::MssqlProvider;
2use std::{env::var, ffi::OsStr};
3use storm::{provider::ProviderFactory, BoxFuture, Error, Result};
4use tiberius::Config;

6pub struct MssqlFactory(pub Config);
7
8impl MssqlFactory {
9    pub fn from_env<K>(var_name: K) -> Result<Self>
10    where
11        K: AsRef<OsStr>,
12    {
13        Ok(Self(Config::from_ado_string(
14            &var(var_name).map_err(Error::std)?,
15        )?))
16    }
17}
18
19impl ProviderFactory for MssqlFactory {
20    type Provider = MssqlProvider;
21
22    fn create_provider(&self) -> BoxFuture<'_, Result<Self::Provider>> {
