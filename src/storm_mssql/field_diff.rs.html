11/// Create the field from the diff value.
12pub trait FromFieldDiff: Sized {
13    fn from_field_diff(value: Value) -> Result<Self>;
14}
15
16pub trait ApplyFieldDiff: Sized {
17    fn apply_field_diff(&mut self, value: Value) -> Result<()>;
18}
19
20pub fn apply_field_diff_impl<T: for<'de> Deserialize<'de>>(v: &mut T, value: Value) -> Result<()> {
21    *v = from_field_diff_impl(value)?;
22    Ok(())
23}
24
25pub trait FieldDiff {
26    fn field_diff(&self, old: &Self) -> Option<Value>;
27}
28
29pub fn field_diff_impl<T: PartialEq + Serialize>(new: &T, old: &T) -> Option<Value> {
30    if new == old {
31        None
32    } else {
33        Some(serde_json::json!(old))
34    }
35}
36
37#[doc(hidden)]
38pub fn _replace_field_diff<K: Eq + Hash, T: ApplyFieldDiff, S: BuildHasher>(
39    field: &mut T,
40    name: K,
41    map: &HashMap<FieldsOrStr<K>, Value, S>,
42) -> Result<()> {
43    if let Some(old) = map.get(&FieldsOrStr::Fields(name)) {
44        field.apply_field_diff(old.clone())?;
45    }
46
47    Ok(())
48}

50impl<T: FromFieldDiff> FromFieldDiff for Option<T> {
51    fn from_field_diff(value: Value) -> Result<Self> {
52        Ok(if value.is_null() {
53            None
54        } else {
55            Some(T::from_field_diff(value)?)
56        })
57    }
58}
59
60impl<T: ApplyFieldDiff + FromFieldDiff> ApplyFieldDiff for Option<T> {
61    fn apply_field_diff(&mut self, value: Value) -> Result<()> {
62        if value.is_null() {
63            *self = None;
64            return Ok(());
65        }
66
67        match self {
68            Some(v) => v.apply_field_diff(value),
69            None => {
70                *self = Some(T::from_field_diff(value)?);
71                Ok(())
72            }
73        }
74    }
75}
76
77impl<T: PartialEq + Serialize> FieldDiff for Option<T> {
78    fn field_diff(&self, old: &Self) -> Option<Value> {
79        field_diff_impl(self, old)
80    }
81}

83pub fn from_field_diff_impl<T: for<'de> Deserialize<'de>>(value: Value) -> Result<T> {
84    serde_json::from_value(value).map_err(Error::std)
85}

87macro_rules! diff {
88    ($n:ty) => {
89        impl ApplyFieldDiff for $n {
90            fn apply_field_diff(&mut self, value: Value) -> Result<()> {
91                apply_field_diff_impl(self, value)
92            }
93        }
94
95        impl FieldDiff for $n {
96            fn field_diff(&self, old: &Self) -> Option<Value> {
97                field_diff_impl(self, old)
98            }
99        }
100
101        impl FromFieldDiff for $n {
102            fn from_field_diff(value: Value) -> Result<Self> {
103                from_field_diff_impl(value)
104            }
105        }
106    };
107}
