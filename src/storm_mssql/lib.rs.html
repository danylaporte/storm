55#[doc(hidden)]
56#[allow(clippy::expect_used)]
57#[allow(clippy::panic)]
58pub fn test_entity<'a>(
59    provider: &'a str,
60    table: &'a str,
61    translated_table: &'a str,
62    expected: &'a [(&'a str, MaxLength)],
63) -> Pin<Box<dyn Future<Output = ()> + 'a>> {
64    Box::pin(async move {
65        let container =
66            crate::create_provider_container_from_env("DB", provider).expect("container");
67        let provider = container
68            .provide::<MssqlProvider>(provider)
69            .await
70            .expect("provider");
71
72        fn split_table_schema(table: &str) -> (&str, &str) {
73            let mut iter = table
74                .split('.')
75                .map(|s| s.trim_matches('[').trim_matches(']'))
76                .filter(|s| !s.is_empty());
77
78            match (iter.next(), iter.next()) {
79                (None, Some(t)) | (Some(t), None) => ("dbo", t),
80                (Some(s), Some(t)) => (s, t),
81                _ => panic!("table invalid"),
82            }
83        }
84
85        let (schema1, table1) = split_table_schema(table);
86        let (schema2, table2) = if translated_table.is_empty() {
87            ("dbo", table1)
88        } else {
89            split_table_schema(translated_table)
90        };
91
92        let sql = r"
93        SELECT
94            c.[name],
95            CAST(c.max_length as int),
96            CAST((CASE WHEN c.system_type_id IN (99, 231, 239) THEN 1 ELSE 0 END) as BIT) IsNChar
97        FROM
98            sys.all_columns c
99            INNER JOIN sys.tables t
100            ON c.object_id = t.object_id
101        WHERE
102            (t.name = @P1 AND t.schema_id = SCHEMA_ID(@P2))
103            OR (t.name = @P3 AND t.schema_id = SCHEMA_ID(@P4))";
104
105        let mut actual: Vec<(String, i32)> = provider
106            .query_rows(
107                sql.to_string(),
108                &[&table1, &schema1, &table2, &schema2],
109                |row| {
110                    let is_nchar = row.get::<bool, _>(2).unwrap_or_default();
111                    let nchar_div = if is_nchar { 2 } else { 1 };
112                    let mut max_length = row.get::<i32, _>(1).unwrap_or_default();
113
114                    if max_length == -1 {
115                        max_length = 0
116                    } else {
117                        max_length /= nchar_div;
118                    }
119
120                    Ok((
121                        row.get::<&str, _>(0).unwrap_or_default().to_lowercase(),
122                        max_length,
123                    ))
124                },
125                false,
126            )
127            .await
128            .expect("rows");
129
130        actual.sort_unstable_by(|a, b| a.0.cmp(&b.0));
131
132        for expected in expected {
133            let n = expected.0.to_lowercase();
134            let n = n.trim_matches('[').trim_matches(']');
135
136            let Some(actual) = actual.iter().find(|t| t.0 == n) else {
137                panic!("column {n} not found");
138            };
139
140            if expected.1 > 0 && expected.1 as i32 != actual.1 {
141                panic!(
142                    "field maxlength {} differ, actual: {}, expected: {}",
143                    &expected.0, actual.1, expected.1
144                );
145            }
146        }
147    })
148}
