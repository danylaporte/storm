pub fn create_provider_container_from_env(env_var: &str, name: &str) -> Result<ProviderContainer> {
    let factory = MssqlFactory::from_env(env_var)?;

    let mut container = ProviderContainer::new();
    container.register(name, factory);

    Ok(container)
}
