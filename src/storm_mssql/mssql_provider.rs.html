30    async fn state(&self) -> MutexGuard<'_, State> {
31        let mut guard = self.0.state.lock().await;
32
33        if self.0.cancel_transaction.swap(false, Relaxed) {
34            let _ = guard.cancel().await;
35        }
36
37        guard
38    }

40    /// Creates a new [Client](Client) instance.
41    /// # Safety
42    /// This operation is safe but the returning client is not constrained by the lock and can modify the database without storm's knowledge.
43    pub async unsafe fn create_client(&self) -> Result<Client> {
44        self.0.state.lock().await.create_client().await
45    }
46
47    /// Creates a new [Client](Client) instance.
48    /// # Safety
49    /// This operation is safe but the returning client is not constrained by the lock and can modify the database without storm's knowledge.
50    pub async unsafe fn get_transaction_client(&self) -> Option<MssqlTransactionGuard<'_>> {
51        let guard = self.0.state.lock().await;
52        guard
53            .transaction
54            .is_some()
55            .then_some(MssqlTransactionGuard(guard))
56    }

58    async fn query_rows_imp<'a, 'b, M, R, C>(
59        &'a self,
60        sql: &'b str,
61        params: &'b [&'b (dyn ToSql)],
62        mut mapper: M,
63        use_transaction: bool,
64    ) -> Result<C>
65    where
66        C: Default + Extend<R> + Send,
67        M: FnMut(Row) -> Result<R> + Send + 'a,
68        R: Send,
69        'a: 'b,
70    {
71        let mut conn = QueryConn::new(self, use_transaction).await?;
72        let mut query = conn.query(sql, params).await?;
73        let mut vec = Vec::with_capacity(10);
74        let mut coll = C::default();
75
76        while let Some(row) = query.try_next().await? {
77            vec.push(mapper(row)?);
78
79            if vec.len() == 10 {
80                #[allow(clippy::iter_with_drain)]
81                coll.extend(vec.drain(..));
82            }
83        }
84
85        if !vec.is_empty() {
86            coll.extend(vec);
87        }
88
89        query.complete().await?;
90        conn.complete();
91
92        Ok(coll)
93    }

95    pub async fn set_client_lock_timeout(&self, timeout: Option<Duration>) -> Result<()> {
96        self.0.state.lock().await.set_lock_timeout(timeout).await
97    }

100impl Execute for MssqlProvider {
101    fn execute_with_args<'a, S>(
102        &'a self,
103        statement: S,
104        params: &'a [&'a (dyn ToSql)],
105        args: ExecuteArgs,
106    ) -> BoxFuture<'a, Result<u64>>
107    where
108        S: Debug + Into<Cow<'a, str>> + Send + 'a,
109    {
110        Box::pin(async move {
111            let mut intermediate = Vec::new();
112            let mut output = Vec::new();
113
114            adapt_params(params, &mut intermediate, &mut output);
115
116            let mut client;
117            let client_ref;
118            let mut guard = self.state().await;
119
120            if args.use_transaction {
121                client = guard.transaction().await?;
122                client_ref = &mut guard.transaction;
123            } else {
124                client = guard.client().await?;
125                client_ref = &mut guard.client;
126            };
127
128            let count = match client.execute(statement, &output).await.map(|v| v.total()) {
129                Ok(count) => count,
130                Err(e) => {
131                    let _ = trace_deadlock(&mut client).await;
132                    return Err(e.into());
133                }
134            };
135
136            *client_ref = Some(client);
137            Ok(count)
138        })
139    }
140}

158        self.0.cancel_transaction.store(true, Relaxed);
159
160        let p = Self(Arc::clone(&self.0));

163            let _ = p.state().await;

172impl QueryRows for MssqlProvider {
173    fn query_rows<'a, S, M, R, C>(
174        &'a self,
175        statement: S,
176        params: &'a [&'a (dyn ToSql)],
177        mut mapper: M,
178        use_transaction: bool,
179    ) -> BoxFuture<'a, Result<C>>
180    where
181        C: Default + Extend<R> + Send,
182        M: FnMut(Row) -> Result<R> + Send + 'a,
183        R: Send,
184        S: Debug + for<'b> Into<Cow<'b, str>> + Send + 'a,
185    {
186        let sql = statement.into();
187
188        Box::pin(async move {
189            let mut count = 0;
190
191            loop {
192                let r = self
193                    .query_rows_imp(&sql, params, &mut mapper, use_transaction)
194                    .await;
195
196                if r.is_ok() || count > 5 {
197                    return r;
198                }
199
200                count += 1;
201            }
202        })
203    }
204}

212impl<'a> QueryConn<'a> {
213    async fn new(provider: &'a MssqlProvider, use_transaction: bool) -> Result<QueryConn<'a>> {
214        let mut guard = provider.state().await;
215
216        let client = match use_transaction {
217            true => guard.transaction().await,
218            false => guard.client().await,
219        }?;
220
221        Ok(Self {
222            client,
223            guard,
224            use_transaction,
225        })
226    }
227
228    fn complete(mut self) {
229        match self.use_transaction {
230            true => self.guard.transaction = Some(self.client),
231            false => self.guard.client = Some(self.client),
232        }
233    }
234
235    async fn query<'b, 'c>(
236        &'b mut self,
237        sql: &'c str,
238        params: &'c [&'c (dyn ToSql)],
239    ) -> Result<QueryStream<'b>>
240    where
241        'b: 'c,
242    {
243        let mut intermediate = Vec::new();
244        let mut output = Vec::new();
245
246        adapt_params(params, &mut intermediate, &mut output);
247
248        let stream = self.client.query(sql, &output[..]).await?;
249
250        Ok(QueryStream(stream))
251    }

254async fn trace_deadlock(client: &mut Client) -> Result<()> {
255    const SQL: &str = r#"
256        SELECT
257            a.session_id,
258            a.start_time,
259            a.[status],
260            a.command,
261            a.blocking_session_id,
262            a.wait_type,
263            a.wait_time,
264            a.open_transaction_count,
265            a.transaction_id,
266            a.total_elapsed_time,
267            definition = CAST(b.text AS VARCHAR(MAX))
268        FROM
269            SYS.DM_EXEC_REQUESTS a
270            CROSS APPLY sys.dm_exec_sql_text(a.sql_handle) b
271        WHERE
272            a.session_id != @@SPID
273            AND a.database_id = DB_ID()
274            AND blocking_session_id != 0
275    "#;
276
277    let deadlocks = client
278        .simple_query(SQL)
279        .await?
280        .into_first_result()
281        .await?
282        .into_iter()
283        .map(|row| {
284            serde_json::json!({
285                "session_id": row.get::<i16, _>(0),
286                "start_time": row.get::<NaiveDateTime, _>(1),
287                "status": row.get::<&str, _>(2).unwrap_or_default(),
288                "command": row.get::<&str, _>(3).unwrap_or_default(),
289                "blocking_session_id": row.get::<i16, _>(4),
290                "wait_type": row.get::<&str, _>(5).unwrap_or_default(),
291                "wait_time": row.get::<i32, _>(6),
292                "open_transaction_count": row.get::<i32, _>(7),
293                "transaction_id": row.get::<i64, _>(8),
294                "total_elapsed_time": row.get::<i32, _>(9),
295                "definition": row.get::<&str, _>(10).unwrap_or_default(),
296            })
297        })
298        .collect::<Vec<_>>();
299
300    if let Ok(json) = serde_json::to_string_pretty(&serde_json::json!(deadlocks)) {
301        info!(json = json, "potential sql deadlocks");
302    }
303
304    Ok(())
305}

316impl Stream for QueryStream<'_> {
317    type Item = Result<tiberius::Row>;
318
319    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
320        loop {
321            return match self.0.poll_next_unpin(cx) {
322                Poll::Pending => Poll::Pending,
323                Poll::Ready(Some(Ok(tiberius::QueryItem::Metadata(_)))) => continue,
324                Poll::Ready(Some(Ok(tiberius::QueryItem::Row(r)))) => Poll::Ready(Some(Ok(r))),
325                Poll::Ready(Some(Err(e))) => Poll::Ready(Some(Err(e.into()))),
326                Poll::Ready(None) => Poll::Ready(None),
327            };
328        }
329    }

332struct State {
333    client: Option<Client>,
334    factory: Box<dyn ClientFactory>,
335    lock_timeout: Option<Duration>,
336    transaction: Option<Client>,
337}
338
339impl State {
340    fn new(factory: Box<dyn ClientFactory>) -> Self {
341        Self {
342            client: None,
343            factory,
344            lock_timeout: Some(DEFAULT_LOCK_TIMEOUT),
345            transaction: None,
346        }
347    }
348
349    async fn cancel(&mut self) -> Result<()> {
350        self.cancel_or_commit("ROLLBACK").await
351    }
352
353    async fn cancel_or_commit(&mut self, statement: &'static str) -> Result<()> {
354        if let Some(mut client) = self.transaction.take() {
355            let r = client.simple_query(statement).await.map_err(Error::Mssql);
356
357            #[cfg(feature = "telemetry")]
358            {
359                metrics::gauge!("storm_mssql_transaction_count").decrement(1.0);
360            }
361
362            r?;
363
364            if self.client.is_none() {
365                self.client = Some(client);
366            }
367        }
368
369        Ok(())
370    }
371
372    async fn client(&mut self) -> Result<Client> {
373        match self.client.take() {
374            Some(c) => Ok(c),
375            None => {
376                if let Some(client) = self
377                    .factory
378                    .under_transaction()
379                    .then(|| self.transaction.take())
380                    .flatten()
381                {
382                    return Ok(client);
383                }
384
385                self.create_client().await
386            }
387        }
388    }
389
390    async fn commit(&mut self) -> Result<()> {
391        self.cancel_or_commit("COMMIT").await
392    }
393
394    async fn create_client(&self) -> Result<Client> {
395        let mut client = self.factory.create_client().await?;
396
397        set_client_lock_timeout(&mut client, self.lock_timeout).await?;
398
399        Ok(client)
400    }
401
402    async fn set_lock_timeout(&mut self, timeout: Option<Duration>) -> Result<()> {
403        if self.lock_timeout != timeout {
404            let mut client = self.client.take();
405
406            if let Some(client) = client.as_mut() {
407                set_client_lock_timeout(client, timeout).await?;
408            }
409
410            if let Some(client) = self.transaction.as_mut() {
411                set_client_lock_timeout(client, timeout).await?;
412            }
413
414            self.client = client;
415        }
416
417        Ok(())
418    }
419
420    async fn transaction(&mut self) -> Result<Client> {
421        match self.transaction.take() {
422            Some(t) => Ok(t),
423            None => {
424                let mut client = self.client().await?;
425
426                let r = client
427                    .simple_query("BEGIN TRAN")
428                    .await
429                    .map_err(Error::Mssql);
430
431                #[cfg(feature = "telemetry")]
432                {
433                    metrics::gauge!("storm_mssql_transaction_count").increment(1.0);
434                }
435
436                r?;
437
438                Ok(client)
439            }
440        }
441    }
442}

444async fn set_client_lock_timeout(client: &mut Client, timeout: Option<Duration>) -> Result<()> {
445    client
446        .simple_query(format!(
447            "SET LOCK_TIMEOUT {};",
448            timeout.map_or(-1, |d| d.as_millis() as i128)
449        ))
450        .await?;
451
452    Ok(())
453}

455fn adapt_params<'a>(
456    input: &'a [&dyn ToSql],
457    intermediate: &'a mut Vec<Parameter<'a>>,
458    output: &mut Vec<&'a dyn tiberius::ToSql>,
459) {
460    intermediate.extend(input.iter().map(|p| Parameter(p.to_sql())));
461    output.extend(intermediate.iter().map(|p| p as &dyn tiberius::ToSql));
